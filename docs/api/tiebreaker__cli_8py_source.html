18def norm(s: str) -> str:
19 return re.sub(r'\s+', ' ', s.strip().casefold())

21def best_name_match(query: str, candidates: list[str]) -> str | None:
22 q = norm(query)
23 for c in candidates:
24 if norm(c) == q:
25 return c
26 m = get_close_matches(query, candidates, n=1, cutoff=0.75)
27 return m[0] if m else None

37def parse_rank_date_col(series: pd.Series) -> pd.Series:
38 s = series.astype(str).str.replace(r'\.0$', '', regex=True)
39 dt1 = pd.to_datetime(s, format="%Y%m%d", errors="coerce")

153def resolve_player_id(hub: DataHub, name_query: str):
154 players = hub.load_players()
155 candidates = players["full_name"].astype(str).tolist()
156 match = best_name_match(name_query, candidates)
157 if not match:
158 return None, None
159 row = players[players["full_name"] == match].iloc[0]
160 return (int(row["player_id"]) if pd.notna(row["player_id"]) else None), str(row["full_name"])
